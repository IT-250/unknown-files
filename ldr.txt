int ldrPin = A0;
int ledPin = 9;

void setup() {
  Serial.begin(9600);
  pinMode(ldrPin, INPUT);
  pinMode(ledPin, OUTPUT);
}

void loop() {
  int ldrValue = analogRead(ldrPin);
  
  if (ldrValue < 500) { // dark
    digitalWrite(ledPin, HIGH);
    Serial.println("Light ON");
  } else {
    digitalWrite(ledPin, LOW);
    Serial.println("Light OFF");
  }
  
  Serial.print("Light intensity: ");
  Serial.println(ldrValue);
  delay(200);
}









#include <ESP8266WiFi.h>

const char* ssid = "your_SSID";
const char* password = "your_PASSWORD";
const char* server = "192.168.x.x";
const int serverPort = 80;

WiFiClient client;
int ldrPin = A0;

void setup() {
  Serial.begin(115200);
  pinMode(ldrPin, INPUT);
  
  WiFi.begin(ssid, password);
  while (WiFi.status() != WL_CONNECTED) {
    delay(500);
    Serial.print(".");
  }
  Serial.println("\nWiFi connected");
}

void loop() {
  int lightIntensity = analogRead(ldrPin);
  
  if (client.connect(server, serverPort)) {
    String postData = "light_intensity=" + String(lightIntensity);
    client.print("POST /insert_ldr.php HTTP/1.1\r\n");
    client.print("Host: " + String(server) + "\r\n");
    client.print("Content-Type: application/x-www-form-urlencoded\r\n");
    client.print("Content-Length: " + String(postData.length()) + "\r\n\r\n");
    client.print(postData);
    client.stop();
  }
  
  delay(10000);
}
