int rainPin = A0;
int ledPin = 9;
int buzzerPin = 8;

void setup() {
  Serial.begin(9600);
  pinMode(rainPin, INPUT);
  pinMode(ledPin, OUTPUT);
  pinMode(buzzerPin, OUTPUT);
}

void loop() {
  int rainValue = analogRead(rainPin);
  
  if (rainValue < 500) { // raining
    digitalWrite(ledPin, HIGH);
    digitalWrite(buzzerPin, HIGH);
    Serial.println("Rain detected!");
  } else {
    digitalWrite(ledPin, LOW);
    digitalWrite(buzzerPin, LOW);
  }
  
  delay(500);
}











#include <ESP8266WiFi.h>

const char* ssid = "your_SSID";
const char* password = "your_PASSWORD";
const char* server = "192.168.x.x";
const int serverPort = 80;

WiFiClient client;
int rainPin = A0;

void setup() {
  Serial.begin(115200);
  pinMode(rainPin, INPUT);
  
  WiFi.begin(ssid, password);
  while (WiFi.status() != WL_CONNECTED) {
    delay(500);
    Serial.print(".");
  }
  Serial.println("\nWiFi connected");
}

void loop() {
  int rainValue = analogRead(rainPin);
  int isRaining = (rainValue < 500) ? 1 : 0;
  
  if (client.connect(server, serverPort)) {
    String postData = "rain_status=" + String(isRaining) + "&raw_value=" + String(rainValue);
    client.print("POST /insert_rain.php HTTP/1.1\r\n");
    client.print("Host: " + String(server) + "\r\n");
    client.print("Content-Type: application/x-www-form-urlencoded\r\n");
    client.print("Content-Length: " + String(postData.length()) + "\r\n\r\n");
    client.print(postData);
    client.stop();
  }
  
  delay(10000);
}
